### 2. 统一错误处理

#### 前端错误处理 (`electron/renderer/src/utils/error-handler.ts`)

```typescript
export async function apiCall<T>(
  fetchFn: () => Promise<Response>,
  errorPrefix: string = '操作'
): Promise<T> {
  try {
    const response = await fetchFn();
    
    if (!response.ok) {
      // HTTP错误处理
      const error = await response.json().catch(() => ({
        detail: `${errorPrefix}失败: HTTP ${response.status}`
      }));
      throw new Error(ErrorHandler.extractMessage(error));
    }
    
    // 返回JSON数据
    return await response.json();
  } catch (error: any) {
    // 网络错误
    if (error.name === 'TypeError' && error.message === 'Failed to fetch') {
      throw new Error('网络连接失败，请检查网络设置');
    }
    throw error;
  }
}
```

**✅ 处理的错误类型**：
- HTTP状态码错误（400、401、403、404、500等）
- 网络连接失败
- 超时错误
- JSON解析错误

### 2. 直播报告 API

#### 历史记录 (`/api/report/live/history`)

**后端实现** (`server/app/api/live_report.py`):

```python
@router.get("/api/report/live/history")
async def get_live_history(
    limit: int = 20,
    db: Session = Depends(get_db)
):
    # 查询历史记录
    reports = db.query(LiveReport).order_by(
        LiveReport.created_at.desc()
    ).limit(limit).all()
    
    return {
        "success": True,
        "data": [report.to_dict() for report in reports]
    }
```

**前端调用** (`electron/renderer/src/pages/dashboard/ReportsPage.tsx`):

```typescript
const loadHistory = async () => {
  try {
    console.log('🔍 [loadHistory] 开始加载历史记录...');
    const res = await fetch(
      `${FASTAPI_BASE_URL}/api/report/live/history?limit=20`
    );
    
    if (!res.ok) {
      throw new Error(`HTTP ${res.status}`);
    }
    
    const data = await res.json();
    console.log('📚 [loadHistory] 响应数据:', data);
    
    if (data?.success && data?.data) {
      setHistory(data.data);  // 更新UI状态
    }
  } catch (error) {
    console.error('❌ [loadHistory] 加载失败:', error);
  }
};
```

**✅ 数据流**：
- 前端请求：`GET /api/report/live/history?limit=20`
- 后端返回：`{ success: true, data: [{ ... }, { ... }] }`
- 前端展示历史记录列表

```python
app.add_middleware(

    allow_methods=["GET", "POST", "PUT", "DELETE", "OPTIONS", "PATCH", "HEAD"],
    allow_headers=["Content-Type", "Authorization", "Accept", ...],
    expose_headers=["*"],
    max_age=3600,
)

# 自定义CORS中间件处理OPTIONS预检请求
@app.middleware("http")
async def cors_preflight_guard(request: Request, call_next):
    if request.method == "OPTIONS":
        response = Response(status_code=200)
        response.headers["Access-Control-Allow-Origin"] = "*"
        # ... 其他CORS头
        return response
    response = await call_next(request)
    response.headers["Access-Control-Allow-Origin"] = "*"
    return response
```
