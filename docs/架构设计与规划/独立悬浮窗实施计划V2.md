## 🎯 核心需求澄清

### 目标用户

**抖音情感陪伴类主播**（偏向擦边）
- ❌ 不懂电脑技术
- ✅ 只会简单操作
- ✅ 需要一键启动
- ✅ 需要悬浮窗在OBS等直播软件上方

### 使用场景

```
主播桌面（单显示器）:
┌────────────────────────────────┐
│  OBS直播软件（全屏）             │
│                                │
│  [直播画面]                     │  ┌──────────┐
│                                │  │ 悬浮窗   │ ← 始终在最上层
│                                │  │ AI建议   │
│                                │  └──────────┘
└────────────────────────────────┘
```

### 核心要求

1. **桌面级悬浮** - 可以移动到桌面任意位置，不受Electron窗口限制
2. **始终置顶** - 覆盖在OBS、抖音等所有应用上方
3. **一键启动** - 点击"开始转写"自动弹出悬浮窗
4. **简单操作** - 主播无需任何额外设置

## 📋 技术方案

### 架构设计

```
┌─────────────────────────────────────┐
│  Electron 主进程 (main.js)           │
│                                     │
│  ┌──────────┐      ┌──────────┐    │
│  │ 主窗口    │      │ 悬浮窗    │    │
│  │ (主程序)  │◄────►│ (独立窗口)│    │
│  └──────────┘  IPC  └──────────┘    │
│       ▲                  ▲          │
└───────┼──────────────────┼──────────┘
        │                  │
        │                  │
    ┌───▼──────────────────▼───┐
    │  渲染进程1      渲染进程2  │
    │  (全功能界面)   (悬浮窗UI) │
    │                           │
    │  共享数据通过IPC同步       │
    └───────────────────────────┘
```

### 关键技术点

1. **独立BrowserWindow**
   - `frame: false` - 无边框
   - `transparent: true` - 透明背景
   - `alwaysOnTop: true` - 始终置顶
   - `skipTaskbar: true` - 不显示在任务栏

2. **IPC通信**
   - 主窗口 → 主进程: `show-floating-window`
   - 主进程 → 悬浮窗: `update-data`
   - 悬浮窗 → 主进程: `close-floating-window`

3. **数据同步**
   - 主窗口监听WebSocket/SSE数据
   - 通过IPC实时推送到悬浮窗
   - 悬浮窗只负责显示，不直接连接后端

### Phase 1: Electron主进程改造

#### 1.1 创建悬浮窗管理模块

**文件**: `electron/floatingWindow.js`

```javascript
const { BrowserWindow } = require('electron');
const path = require('path');

let floatingWindow = null;

/**
 * 创建独立悬浮窗
 */
function createFloatingWindow(rendererURL) {
  if (floatingWindow) {
    floatingWindow.show();
    return floatingWindow;
  }
  
  floatingWindow = new BrowserWindow({
    width: 320,
    height: 280,
    frame: false,           // 无边框
    transparent: true,      // 透明背景
    alwaysOnTop: true,     // 始终置顶
    skipTaskbar: true,     // 不显示在任务栏
    resizable: false,      // 不可调整大小
    hasShadow: true,       // 阴影效果
    webPreferences: {
      nodeIntegration: false,
      contextIsolation: true,
      preload: path.join(__dirname, 'preload.js')
    }
  });
  
  // 加载悬浮窗页面
  const isDev = !require('electron').app.isPackaged;
  if (isDev) {
    floatingWindow.loadURL(`${rendererURL}/#/floating`);
  } else {
    floatingWindow.loadFile(
      path.join(__dirname, 'renderer', 'dist', 'index.html'),
      { hash: '/floating' }
    );
  }
  
  // 窗口关闭时清理引用
  floatingWindow.on('closed', () => {
    floatingWindow = null;
  });
  
  return floatingWindow;
}

/**
 * 显示悬浮窗
 */
function showFloatingWindow() {
  if (floatingWindow) {
    floatingWindow.show();
    floatingWindow.focus();
  }
}

/**
 * 隐藏悬浮窗
 */
function hideFloatingWindow() {
  if (floatingWindow) {
    floatingWindow.hide();
  }
}

/**
 * 关闭悬浮窗
 */
function closeFloatingWindow() {
  if (floatingWindow) {
    floatingWindow.close();
    floatingWindow = null;
  }
}

/**
 * 发送数据到悬浮窗
 */
function sendDataToFloating(channel, data) {
  if (floatingWindow && !floatingWindow.isDestroyed()) {
    floatingWindow.webContents.send(channel, data);
  }
}

/**
 * 获取悬浮窗实例
 */
function getFloatingWindow() {
  return floatingWindow;
}

module.exports = {
  createFloatingWindow,
  showFloatingWindow,
  hideFloatingWindow,
  closeFloatingWindow,
  sendDataToFloating,
  getFloatingWindow
};
```

#### 4.2 悬浮窗位置持久化

**实现方式**:

```javascript
// electron/floatingWindow.js

function createFloatingWindow(rendererURL) {
  // 从配置读取上次位置
  const savedPosition = loadFloatingPosition();
  
  floatingWindow = new BrowserWindow({
    x: savedPosition.x || undefined,
    y: savedPosition.y || undefined,
    // ... 其他配置
  });
  
  // 监听位置变化
  floatingWindow.on('moved', () => {
    const [x, y] = floatingWindow.getBounds();
    saveFloatingPosition({ x, y });
  });
}

function loadFloatingPosition() {
  try {
    const config = fs.readFileSync('config/floating-position.json', 'utf8');
    return JSON.parse(config);
  } catch {
    return { x: null, y: null };
  }
}

function saveFloatingPosition(position) {
  try {
    fs.writeFileSync(
      'config/floating-position.json',
      JSON.stringify(position)
    );
  } catch (e) {
    console.error('保存位置失败:', e);
  }
}
```

### Phase 5: 边缘吸附实现

**方式**: 使用Electron窗口事件

```javascript
floatingWindow.on('move', () => {
  const bounds = floatingWindow.getBounds();
  const display = screen.getPrimaryDisplay();
  const { width, height } = display.workAreaSize;
  
  const SNAP_THRESHOLD = 30;
  let newX = bounds.x;
  let newY = bounds.y;
  
  // 左边缘吸附
  if (bounds.x < SNAP_THRESHOLD) {
    newX = 0;
  }
  // 右边缘吸附
  else if (width - (bounds.x + bounds.width) < SNAP_THRESHOLD) {
    newX = width - bounds.width;
  }
  
  // 上下边缘同理
  
  if (newX !== bounds.x || newY !== bounds.y) {
    floatingWindow.setBounds({ x: newX, y: newY });
  }
});
```
